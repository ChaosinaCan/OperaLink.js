 17 /*
 18  * This library facilitates making requests to the Opera Link synchronization
 19  * server. It is intended for use in Opera extensions, but it can be used in any
 20  * situation where JavaScript security allows cross domain XML HTTP requests.
 21  * 
 22  * This library depends on oauth.js and sha1.js, which can be found here:
 23  * http://oauth.googlecode.com/svn/code/javascript/
 24  * 
 25  * Before sending any requests to Opera Link, you must first authenticate with
 26  * OAuth. To do this, your application needs a consumer key and secret, which
 27  * you can obtain by registering your application at
 28  * https://auth.opera.com/service/oauth/applications/
 29  * 
 30  * Call opera.link.consumer() with your application's consumer key and secret.
 31  * Now, each user needs their own token and token secret. A full explanation of
 32  * the authentication process can be found at
 33  * http://dev.opera.com/articles/view/gentle-introduction-to-oauth/
 34  * 
 35  * This library simplifies the process somewhat. Call opera.link.requestToken()
 36  * to get a request token and token secret (steps 1-5). Save these values so
 37  * that you can recall them once the user grants access to your application. 
 38  * This function will automatically send the user to the authorization page if
 39  * used in an Opera extension. If the extensions API is not available, you must
 40  * set opera.link.authorizeFunction and handle the opening of the authorization
 41  * page yourself.
 42  * 
 43  * Once the user grants access to your application (step 6), they will get a 
 44  * 6-digit verifier code (step 7). Call opera.link.getAccessToken with the 
 45  * request token, token secret, and the verifier to get an access token and 
 46  * token secret (steps 8,9). 
 47  * You should save the access token and token secret to permanent storage so
 48  * they can be recalled later and the user will not have to repeat the 
 49  * authentication process.
 50  * 
 51  * To set up authentication from a saved access token, call 
 52  * opera.link.authorize() with the saved token and token secret. Next, check 
 53  * that the grant has not expired by calling opera.link.testAuthorization(). If
 54  * the grant has expired, you will need to redo the full authentication process.
 55  * 
 56  * 
 57  * opera.link provides methods to set up authentication and send OAuth signed
 58  *		requests.
 59  *		
 60  *	opera.link.utils provides generic methods and methods common to all datatypes
 61  *	
 62  *	opera.link.bookmarks, opera.link.notes, opera.link.searchengines,
 63  *	opera.link.speeddial, and opera.link.urlfilter provide access to each of the
 64  *		data types supported by Opera Link.
 65  *		
 66  *	Documentation on the parameters of specific request types can be found here:
 67  *	http://www.opera.com/docs/apis/linkrest/
 68  *	
 69  *	Examples using the Opera Link API can be found here:
 70  *	http://dev.opera.com/articles/view/introducing-the-opera-link-api/
 71  */

 81 /**
 82  * @namespace Enables communication with the Opera Link server.
 83  * @requires Add access to https://auth.opera.com and https://link.api.opera.com 
 84  * in your extension's config.xml.
 85  */
 86 opera.link = new function OperaLink() {
 87 	
 88 	/**
 89 	 * @class A list of the response codes used by Opera Link
 90 	 * @static
 91 	 */
 92 	this.response = {
 93 		/**
 94 		 * 200: The request completed successfully.
 95 		 * @constant
 96 		 */
 97 		Ok: 200,
 98 		/**
 99 		 * 204 : The item was successfully deleted.
100 		 * @constant
101 		 */
102 		Deleted: 204,
103 		/**
104 		 * 400 : The request is invalid and cannot be processed. The cause of this is
105 		 * often missing a required parameter or trying to execute an invalid
106 		 * method on an item.
107 		 * @constant
108 		 */
109 		BadRequest: 400,
110 		/**
111 		 * 401 : The request cannot be allowed, possibly because your authentication
112 		 * information is invalid or because too many requests sent in a short
113 		 * period made the throttling ban them.
114 		 * @constant
115 		 */
116 		Unauthorized: 401,
117 		/**
118 		 * 404 : The item you seek or wish to manipulate was not found
119 		 * @constant
120 		 */
121 		NotFound: 404,
122 		/**
123 		 * 405 : The method you tried to use is not allowed
124 		 * @constant
125 		 */
126 		MethodNotAllowed: 405,
127 		/**
128 		 * 500 : This is an unexpected server error.
129 		 * @constant
130 		 */
131 		InternalServerError: 500,
132 		/**
133 		 * 501 : You are trying to execute a method that is not implemented. This can
134 		 * happen when you execute a method that is not supported by the specific
135 		 * datatype or if you misspelled a method name in the request.
136 		 * @constant
137 		 */
138 		NotImplemented: 501
139 	}
140 	
141 	/**
142 	 * The location of the Opera Link REST API
143 	 * @type String
144 	 */
145 	this.apiurl = 'https://link.api.opera.com/rest/';
146 	
147 	/**
148 	 * If true, the results of actions that only return one object will be 
149 	 * simplified from an array containing one object to just the one object.
150 	 * Defaults to true.
151 	 * @type Boolean
152 	 */
153 	this.simplifyResults = true;
154 	
155 	/**
156 	 * If using this library outside of extensions, opera.link.requestToken will
157 	 * call this function instead of opera.extension.tabs.create to show the user
158 	 * the authorization page. The function should take one parameter, the url
159 	 * of the authorization page.
160 	 * @type Function
161 	 */
162 	this.authorizeFunction = null;
163 	
164 	/**
165 	 * Sets the storage object used by saveToken and loadToken
166 	 * @type Storage
167 	 */
168 	this.storage = (widget && widget.preferences) ? widget.preferences : localStorage;
169 	
170 	
171 	/**
172 	 * Authentication parameters for OAuth
173 	 * @private
174 	 */
175 	var accessor = {
176 		consumerKey : null,
177 		consumerSecret: null,
178 		token: null,
179 		tokenSecret: null,
180 	}
181 
182 	/**
183 	 * Parameters used for OAuth authentication. Don't change these unless you 
184 	 * know what you're doing.
185 	 */
186 	this.provider = {
187 		signatureMethod: 'HMAC-SHA1',
188 		requestTokenURL: 'https://auth.opera.com/service/oauth/request_token',
189 		userAuthorizationURL: 'https://auth.opera.com/service/oauth/authorize',
190 		accessTokenURL: 'https://auth.opera.com/service/oauth/access_token',
191 	}
192 	
193 	
194 	/**
195 	 * Sends an OAuth GET request to the specified URL
196 	 * @param {String} url The location to send the request
197 	 * @param {Object} params The data to send. Use null for no data
198 	 * @param {Function(xhr)} callback Function called when the request completes.
199 	 *		The callback is passed one argument: the XMLHttpRequest object used
200 	 */
201 	this.get = function(url, params, callback) {
202 		var message = {
203 			action: url,
204 			method: 'GET',
205 			parameters: params
206 		}
207 		
208 		OAuth.completeRequest(message, accessor);
209 		url += '?' + OAuth.formEncode(message.parameters);
210 		
211 		var xhr = new XMLHttpRequest();
212 		xhr.onreadystatechange = function() {
213 			if (xhr.readyState == 4)
214 				callback(xhr);
215 		}
216 	
217 		xhr.open(message.method, url, true);
218 		xhr.send(null);
219 	}
220 	
221 	/**
222 	 * Sends an OAuth POST request to the specified URL
223 	 * @param {String} url The location to send the request
224 	 * @param {Object} params The data to send. Use null for no data
225 	 * @param {Function(xhr)} callback Function called when the request completes.
226 	 *		The callback is passed one argument: the XMLHttpRequest object used.
227 	 */
228 	this.post = function(url, params, callback) {
229 		var message = {
230 			action: url,
231 			method: 'POST',
232 			parameters: null	// parameters do not get signed when posting JSON data
233 		}
234 		
235 		var requestBody = JSON.stringify(params);
236 		OAuth.completeRequest(message, accessor);
237 		var authorizationHeader = OAuth.getAuthorizationHeader('', message.parameters);
238 		
239 		var xhr = new XMLHttpRequest();
240 		xhr.onreadystatechange = function() {
241 			if (xhr.readyState == 4)
242 				callback(xhr);
243 		}
244 	
245 		xhr.open(message.method, message.action, true);
246 		xhr.setRequestHeader('Authorization', authorizationHeader);
247 		xhr.setRequestHeader('Content-Type', 'application/json');
248 		xhr.send(requestBody);
249 	}
250 	
251 	/**
252 	 * Sets the OAuth consumer key and secret. These are given to you when you
253 	 * set up a new application with Opera Link and are specific to your application.
254 	 * @param {String} key The application's consumer key
255 	 * @param {String} secret The application's consumer secret
256 	 */
257 	this.consumer = function(key, secret) {
258 		accessor.consumerKey = key;
259 		accessor.consumerSecret = secret;
260 	}
261 	
262 	/**
263 	 * Sets the OAuth access token and token secret. These are specific to each user.
264 	 * @param {String} token The user's access token
265 	 * @param {String} secret The user's access token secret
266 	 */
267 	this.authorize = function(token, secret) {
268 		accessor.token = token;
269 		accessor.tokenSecret = secret;
270 	}
271 	
272 	/**
273 	 * Requests a new request token. This will open a new tab where the user can
274 	 * grant access to your application. The resulting request token, token secret,
275 	 * and 6-digit verifier must be used with opera.link.getAccessToken to get a
276 	 * permanent access token.
277 	 * @param {Function(data)} callback Function that is called if the request
278 	 *		succeeds. The callback is passed one argument, an object with two
279 	 *		properties: "token", the temporary request token, and "secret", the
280 	 *		secret that goes with the token.
281 	 * @param {Function(xhr)} [error] Function that is called if the request fails. 
282 	 *		The function is passed one argument: the XMLHttpRequest object used.
283 	 */
284 	this.requestToken = function(callback, error) {
285 		this.getRequestToken(function success(e) {
286 			opera.link.authorizeRequestToken(e.token);
287 			callback(e);
288 		}, error);
289 	}
290 	
291 	/**
292 	 * Requests a new access token.
293 	 * @param {String} requestToken The request token
294 	 * @param {String} requestSecret The request token secret
295 	 * @param {String} verifier The 6-digit verifier code
296 	 * @param {Function(data)} callback Function that is called if the request
297 	 *		succeeds. The callback is passed one argument, an object with two
298 	 *		properties: "token", the access token, and "secret", the secret that 
299 	 *		goes with the token.
300 	 * @param {Function(xhr)} [error] Function that is called if the request fails. 
301 	 *		The function is passed one argument: the XMLHttpRequest object used.
302 	 */
303 	this.getAccessToken = function(requestToken, requestSecret, verifier, callback, error) {
304 		this.authorize(requestToken, requestSecret);
305 		
306 		this.get(this.provider.accessTokenURL, {
307 			'oauth_signature_method': this.provider.signatureMethod,
308 			'oauth_verifier': verifier,
309 		}, function(xhr) {
310 			if (xhr.status == 200) {
311 				var params = parseResponse(xhr.responseText);
312 				opera.link.authorize(params.oauth_token, params.oauth_token_secret);
313 				callback({token: params.oauth_token, secret: params.oauth_token_secret});
314 			}
315 			else if (error)
316 				error(xhr);
317 		});
318 	}
319 	
320 	/**
321 	 * Tests whether the current authentication tokens are valid
322 	 * @param {Function(success)} callback Function which is called with the 
323 	 *		result of the test. The function is passed one argument: true if 
324 	 *		the authorization parameters are correct or false otherwise.
325 	 */
326 	this.testAuthorization = function(callback) {
327 		this.get(this.apiurl + 'bookmark', null, function(xhr) {
328 			callback(xhr.status != opera.link.response.Unauthorized);
329 		});
330 	}
331 	
332 	
333 	
334 	
335 	this.getRequestToken = function(callback, error) {
336 		this.get(this.provider.requestTokenURL, {
337 				'oauth_signature_method': this.provider.signatureMethod,
338 				'oauth_callback': 'oob',
339 			}, function (xhr) {		
340 				if (xhr.status == 200) {
341 					var params = parseResponse(xhr.responseText);
342 					callback({token: params.oauth_token, secret: params.oauth_token_secret});
343 				}
344 				else if (error)
345 					error(xhr);
346 			}
347 		);
348 	}
349 	
350 	this.authorizeRequestToken = function(requestToken) {
351 		var message = {
352 			action: this.provider.userAuthorizationURL,
353 			method: 'GET',
354 			parameters: {
355 				'oauth_signature_method': this.provider.signatureMethod,
356 				'oauth_callback': 'oob',
357 				'oauth_token': requestToken,
358 			}
359 		}
360 		
361 		OAuth.completeRequest(message, accessor);
362 		var url = message.action + '?' + OAuth.formEncode(message.parameters);
363 		
364 		if (this.authorizeFunction)
365 			this.authorizeFunction(url)
366 		else
367 			opera.extension.tabs.create({url: url, focused: true});
368 	}
369 	
370 	/**
371 	 * Saves the current OAuth token and token secret to storage in the values
372 	 * oauth_token and oauth_secret.
373 	 */
374 	this.saveToken = function() {
375 		this.storage['oauth_token'] = JSON.stringify(accessor.token);
376 		this.storage['oauth_secret'] = JSON.stringify(accessor.tokenSecret);
377 	}
378 	
379 	/**
380 	 * Loads the OAuth token and token secret stored in the values oauth_token 
381 	 * and oauth_secret.
382 	 * @returns {Boolean} True if there were token values to load, false otherwise
383 	 */
384 	this.loadToken = function() {
385 		var token = JSON.parse(this.storage['oauth_token'] || null);
386 		var secret = JSON.parse(this.storage['oauth_secret'] || null);
387 		
388 		if (token && secret) {
389 			this.authorize(token, secret);
390 			return true;
391 		}
392 		return false;
393 	}
394 
395 	/**
396 	 * Deletes the saved OAuth token and token secret from storage
397 	 */
398 	this.clearSavedToken = function() {
399 		this.storage.removeItem('oauth_token');
400 		this.storage.removeItem('oauth_secret');
401 	}
402

404 	/**
405 	 * Parses the response of a token request
406 	 * @private
407 	 */
408 	var parseResponse = function(q) {
409 		var items = q.split('&');
410 		var result = {};
411 		for (var i = 0; i < items.length; i++) {
412 			var temp = items[i].split('=');
413 			var key = decodeURIComponent(temp[0].replace(/\+/g, '%20'));
414 			var value = decodeURIComponent(temp[1].replace(/\+/g, '%20'));
415 			result[key] = value;
416 		}
417 		return result;
418 	}
419 }

422 /**
423  * @namespace Utility methods and methods common to all datatypes
424  */
425 opera.link.util = new function OperaLinkUtils() {
426 	
427 	/**
428 	 * Gets the properties of the requested item
429 	 * @param {String} datatype The item type
430 	 * @param {String} item The item id
431 	 * @param {null|Object} params Extra parameters for the request. 
432 	 *		Use null if no parameters are needed.
433 	 * @param {Function(data)} callback Function which is called with the result
434 	 *		of the request. The function is passed one argument, an object with 
435 	 *		two properties: "status", the response code, and "response", the JSON 
436 	 *		parsed response body.
437 	 */
438 	this.get = function(datatype, item, params, callback) {
439 		var url = opera.link.apiurl + datatype + '/' + item;
440 		if (url[url.length - 1] != '/')
441 			url += '/';
442 		
443 		opera.link.get(url, completeParams(params), function(xhr) {
444 			var response = xhr.status == opera.link.response.Ok ? 
445 				JSON.parse(xhr.responseText || 'null') : xhr.responseText;
446 			callback({
447 				status: xhr.status,
448 				response: response,
449 			});
450 		});
451 	}
452 	
453 	/**
454 	 * Sends a POST request to Opera Link
455 	 * @param {String} method One of the following: create, delete, trash, update, or move
456 	 * @param {String} datatype The item type
457 	 *	@param {null|String} item The id of the item to modify. Use null if not applicable
458 	 * @param {Object} params Parameters for the request. Required parameters
459 	 *		depend on the method and datatype used
460 	 *	@param {Function(data)} callback Function which is called with the result
461 	 *		of the request. The function is passed one argument, an object with 
462 	 *		two properties: "status", the response code, and "response", the JSON 
463 	 *		parsed response body.
464 	 */
465 	this.post = function(method, datatype, item, params, callback) {
466 		var url = opera.link.apiurl + datatype + '/' + (item || '');
467 		if (url[url.length - 1] != '/')
468 			url += '/';
469 		
470 		params = completeParams(params);
471 		params.api_method = method;
472 		
473 		opera.link.post(url, params, function(xhr) {
474 			var response = xhr.status == opera.link.response.Ok ? 
475 				JSON.parse(xhr.responseText || 'null') : xhr.responseText;
476 			callback({
477 				status: xhr.status,
478 				response: response,
479 			});
480 		});
481 	}
482 	
483 	/**
484 	 * Returns an image data URL suitable for use when creating bookmarks
485 	 * @param {HTMLImageElement} image The image to convert
486 	 * @param {Number} [size] The size of the icon (defaults to 16)
487 	 */
488 	this.makeIcon = function(image, size) {
489 		size = size || 16;
490 		
491 		var canvas = document.createElement('canvas');
492 		canvas.width = size;
493 		canvas.height = size;
494 		var ctx = canvas.getContext('2d');
495 		ctx.drawImage(image, 0, 0, size, size);
496 		return canvas.toDataURL('image/png');
497 	}
498 	
499 	/**
500 	 * Returns an image data URL suitable for use when creating bookmarks
501 	 * Make sure to allow access to the image in config.xml or this function will
502 	 * fail silently without calling the callback function.
503 	 * @param {String} src The location of the icon
504 	 * @param {Function(dataurl)} callback A function which will be called with 
505 	 *		the result of the request. The function should is passed one argument, 
506 	 *		either the icon encoded as a data URL, or null if the request failed
507 	 */
508 	this.getIcon = function(src, callback) {
509 		var img = new Image();
510 		img.onload = function() {
511 			callback(opera.link.util.makeIcon(img));
512 		}
513 		img.onerror = function() {
514 			callback(null);
515 		}
516 		
517 		img.src = src;
518 	}
519 	
520 	
521 	/**
522 	 * Helper function to simplify results of requests that only return one item
523 	 */
524 	this.simplify = function(data, callback) {
525 		if (opera.link.simplifyResults && data.response.length == 1) 
526 			data.response = data.response[0];
527 		
528 		callback(data);
529 	}
530 	
531 	
532 	/**
533 	 * @private
534 	 */
535 	var completeParams = function(params) {
536 		if (!params)
537 			params = {};
538 		
539 		params.api_output = 'json';
540 		return params;
541 	}
542 }

556 	 * @param {Function(result)} callback Function which is called with the result

568 	 * @param {Function(result)} callback Function which is called with the result

575 		item = item.replace('//', '/');
576 		util.get(type, item, null, callback);

582 	 * @param {null|String} parent The parent folder's id
583 	 * @param {Function(result)} callback Function which is called with the result

598 	 * @param {null|String} parent The parent folder's id
599 	 * @param {Function(result)} callback Function which is called with the result
600 	 *		of the request. The function is passed one argument, an object with 
601 	 *		two properties: "status", the response code, and "response", the JSON 
602 	 *		parsed response body.
603 	 */
604 	this.createFolder = function(params, parent, callback) {
605 		params.item_type = 'bookmark_folder';
606 		util.post('create', type, parent, params, function(data) {

621 		parent = parent || null;

629 	 * @param {String} item The item's id
630 	 * @param {Function(result)} callback Function which is called with the result
631 	 *		of the request. The function is passed one argument, an object with

639 	/**
640 	 * Sends a bookmark, folder, or separator to the trash
641 	 * @param {String} item The item's id
642 	 * @param {Function(result)} callback Function which is called with the result
643 	 *		of the request. The function is passed one argument, an object with 
644 	 *		two properties: "status", the response code, and "response", the JSON
